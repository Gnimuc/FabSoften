17inline void dynamicMeanFilter(const cv::Mat &src, cv::Mat &dst, const cv::Mat &radius) {
18 CV_Assert(src.depth() == CV_32F);
19 cv::Mat integralImg;
20 cv::integral(src, integralImg);
21 const auto nRow = integralImg.rows, nCol = integralImg.cols;
22 for (size_t x = 0; x < src.rows; ++x)
23 for (size_t y = 0; y < src.cols; ++y) {
24 const auto r = radius.at<float>(x, y);
25 const auto xA = std::max<int>(x - r, 0);
26 const auto yA = std::max<int>(y - r, 0);
27 const auto &sA = integralImg.at<double>(xA, yA);
28 const auto xB = std::max<int>(x - r, 0);
29 const auto yB = std::min<int>(y + r + 1, nCol - 1);
30 const auto &sB = integralImg.at<double>(xB, yB);
31 const auto xC = std::min<int>(x + r + 1, nRow - 1);
32 const auto yC = std::max<int>(y - r, 0);
33 const auto &sC = integralImg.at<double>(xC, yC);
34 const auto xD = std::min<int>(x + r + 1, nRow - 1);
35 const auto yD = std::min<int>(y + r + 1, nCol - 1);
36 const auto &sD = integralImg.at<double>(xD, yD);
37 dst.at<float>(x, y) = (sD + sA - sB - sC) / ((2 * r + 1) * (2 * r + 1));
38 }
